TABLE MỚI
CREATE DATABASE IF NOT EXISTS engjoy CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE engjoy;


CREATE TABLE IF NOT EXISTS User (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL,
    full_name VARCHAR(100),
    phone_number VARCHAR(20),
    description TEXT,
    birthdate DATE,
    registration_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    user_type TINYINT(1) DEFAULT 1 COMMENT '1=user, 2=admin',


    
CHECK (
       CHAR_LENGTH(password) >= 6 AND
       password REGEXP '[A-Z]' AND
       password REGEXP '[0-9]'
   ),

   CHECK (email LIKE '%@%'),

   CHECK (
       phone_number IS NULL OR 
       phone_number REGEXP '^\\+[0-9]{6,}$'
   ),

   CHECK (user_type IN (1, 2))
	);


CREATE TABLE IF NOT EXISTS Tag (
    tag_id INT AUTO_INCREMENT PRIMARY KEY,
    tag_name VARCHAR(255) NOT NULL UNIQUE
);
CREATE TABLE IF NOT EXISTS WordSet (
    word_set_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100) DEFAULT 'Untitled',
    description TEXT,
    created_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    last_update DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_hidden TINYINT(1) DEFAULT 2,
    user_id INT NOT NULL,


    CHECK (is_hidden IN (1, 2)),


    FOREIGN KEY (user_id) REFERENCES User(user_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
CREATE TABLE IF NOT EXISTS WordSetTag (
    word_set_id INT NOT NULL,
    tag_id INT NOT NULL,
    PRIMARY KEY (word_set_id, tag_id),


    FOREIGN KEY (word_set_id) REFERENCES WordSet(word_set_id)
        ON DELETE CASCADE ON UPDATE CASCADE,


    FOREIGN KEY (tag_id) REFERENCES Tag(tag_id)
        ON DELETE CASCADE ON UPDATE CASCADE
);


CREATE TABLE IF NOT EXISTS SavedWordSet (
    word_set_id INT NOT NULL,
    user_saved_id INT NOT NULL,
    saved_date DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,


    PRIMARY KEY (word_set_id, user_saved_id),


    FOREIGN KEY (word_set_id) REFERENCES WordSet(word_set_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,


    FOREIGN KEY (user_saved_id) REFERENCES User(user_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);






CREATE TABLE IF NOT EXISTS Vocabulary (
    vocab_id INT AUTO_INCREMENT PRIMARY KEY,
    word_set_id INT NOT NULL,
    term VARCHAR(50) NOT NULL,
    definition VARCHAR(50) NOT NULL,


    FOREIGN KEY (word_set_id) REFERENCES WordSet(word_set_id)
        ON DELETE CASCADE ON UPDATE CASCADE
);
CREATE TABLE IF NOT EXISTS Game (
    game_id INT AUTO_INCREMENT PRIMARY KEY,
    game_name VARCHAR(100) NOT NULL UNIQUE
);
CREATE TABLE IF NOT EXISTS GameMode (
    game_mode_id INT AUTO_INCREMENT PRIMARY KEY,
    game_id INT NOT NULL,
    grid_size VARCHAR(100) NOT NULL,
    min_words INT NOT NULL,
    max_words INT NOT NULL,


    FOREIGN KEY (game_id) REFERENCES Game(game_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);


CREATE TABLE Leaderboard (
    leaderboard_id INT DEFAULT 1,
    user_id INT NOT NULL,
    word_set_quantity INT DEFAULT 0,
    created_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (user_id, created_date),
    FOREIGN KEY (user_id) REFERENCES User(user_id)
 ON DELETE CASCADE
 ON UPDATE CASCADE
);




CREATE TABLE IF NOT EXISTS ReportedSet (
    report_id INT AUTO_INCREMENT PRIMARY KEY,
    word_set_id INT NOT NULL,
    report_status TINYINT(1) DEFAULT 0 CHECK (report_status IN (0, 1, 2)),
    reason TEXT,
    reported_date DATETIME DEFAULT CURRENT_TIMESTAMP,


    FOREIGN KEY (word_set_id) REFERENCES WordSet(word_set_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
CREATE TABLE IF NOT EXISTS GameSession (
    session_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    word_set_id INT NOT NULL,
    play_date DATETIME DEFAULT CURRENT_TIMESTAMP,
    play_duration BIGINT,
    end_date DATETIME,
    game_mode_id INT NOT NULL,
    game_theme TINYINT(1) DEFAULT 1 CHECK (game_theme IN (1, 2)),
    game_outline TINYINT(1) DEFAULT 1 CHECK (game_outline IN (1, 2)),
    game_music TINYINT(1) DEFAULT 1 CHECK (game_music IN (1, 2)),
    score INT DEFAULT 0 CHECK (score >= 0),


    FOREIGN KEY (user_id) REFERENCES User(user_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (word_set_id) REFERENCES WordSet(word_set_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (game_mode_id) REFERENCES GameMode(game_mode_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);






TRIGGER MỚI
USE engjoy;
DELIMITER $$


-- Trigger 1: Chỉ user_type = 1 mới được tạo WordSet
CREATE TRIGGER trg_before_insert_wordset_check_user_type
BEFORE INSERT ON WordSet
FOR EACH ROW
BEGIN
    DECLARE role TINYINT;
    SELECT user_type INTO role FROM User WHERE user_id = NEW.user_id;
    IF role <> 1 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Only regular users (user_type = 1) can create WordSets.';
    END IF;
END$$


-- Trigger 2: Tự động cập nhật last_update khi update WordSet
CREATE TRIGGER trg_before_update_wordset_last_update
BEFORE UPDATE ON WordSet
FOR EACH ROW
BEGIN
    SET NEW.last_update = NOW();
END$$




-- Trigger 4: Chỉ user_type = 1 mới được tạo GameSession
CREATE TRIGGER trg_before_insert_gamesession_check_user
BEFORE INSERT ON GameSession
FOR EACH ROW
BEGIN
    DECLARE role TINYINT;
    SELECT user_type INTO role FROM User WHERE user_id = NEW.user_id;
    IF role <> 1 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Only regular users (user_type = 1) can create GameSessions.';
    END IF;
END$$


-- Trigger 5: end_date phải sau play_date khi insert GameSession
CREATE TRIGGER trg_before_insert_gamesession_check_end_date
BEFORE INSERT ON GameSession
FOR EACH ROW
BEGIN
    IF NEW.end_date IS NOT NULL AND NEW.end_date <= NEW.play_date THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'end_date must be after play_date';
    END IF;
END$$


-- Trigger 6: end_date phải sau play_date khi update GameSession
CREATE TRIGGER trg_before_update_gamesession_check_end_date
BEFORE UPDATE ON GameSession
FOR EACH ROW
BEGIN
    IF NEW.end_date IS NOT NULL AND NEW.end_date <= NEW.play_date THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'end_date must be after play_date';
    END IF;
END$$


-- Trigger 7: Cập nhật Leaderboard sau khi insert WordSet
CREATE TRIGGER trg_after_insert_wordset_update_leaderboard
AFTER INSERT ON WordSet
FOR EACH ROW
BEGIN
    IF EXISTS (SELECT 1 FROM User WHERE user_id = NEW.user_id AND user_type = 1) THEN
        UPDATE Leaderboard
        SET word_set_quantity = (
            SELECT COUNT(DISTINCT ws.word_set_id)
            FROM WordSet ws
            JOIN Vocabulary v ON ws.word_set_id = v.word_set_id
            WHERE ws.user_id = NEW.user_id
            GROUP BY ws.user_id
            HAVING COUNT(v.vocab_id) > 5
        ),
        created_date = NOW()
        WHERE user_id = NEW.user_id;
    END IF;
END$$


-- Trigger 8: Cập nhật Leaderboard sau khi xóa WordSet
CREATE TRIGGER trg_after_delete_wordset_update_leaderboard
AFTER DELETE ON WordSet
FOR EACH ROW
BEGIN
    UPDATE Leaderboard
    SET word_set_quantity = (
        SELECT COUNT(DISTINCT ws.word_set_id)
        FROM WordSet ws
        JOIN Vocabulary v ON ws.word_set_id = v.word_set_id
        WHERE ws.user_id = OLD.user_id
        GROUP BY ws.user_id
        HAVING COUNT(v.vocab_id) > 5
    ),
    created_date = NOW()
    WHERE user_id = OLD.user_id;
END$$


-- Trigger 9: Cập nhật Leaderboard sau khi thêm Vocabulary
CREATE TRIGGER trg_after_insert_vocab_update_leaderboard
AFTER INSERT ON Vocabulary
FOR EACH ROW
BEGIN
    DECLARE owner_id INT;
    SELECT user_id INTO owner_id FROM WordSet WHERE word_set_id = NEW.word_set_id;
    UPDATE Leaderboard
    SET word_set_quantity = (
        SELECT COUNT(DISTINCT ws.word_set_id)
        FROM WordSet ws
        JOIN Vocabulary v ON ws.word_set_id = v.word_set_id
        WHERE ws.user_id = owner_id
        GROUP BY ws.user_id
        HAVING COUNT(v.vocab_id) > 5
    ),
    created_date = NOW()
    WHERE user_id = owner_id;
END$$


-- Trigger 10: Cập nhật Leaderboard sau khi xóa Vocabulary
CREATE TRIGGER trg_after_delete_vocab_update_leaderboard
AFTER DELETE ON Vocabulary
FOR EACH ROW
BEGIN
    DECLARE owner_id INT;
    SELECT user_id INTO owner_id FROM WordSet WHERE word_set_id = OLD.word_set_id;
    UPDATE Leaderboard
    SET word_set_quantity = (
        SELECT COUNT(DISTINCT ws.word_set_id)
        FROM WordSet ws
        JOIN Vocabulary v ON ws.word_set_id = v.word_set_id
        WHERE ws.user_id = owner_id
        GROUP BY ws.user_id
        HAVING COUNT(v.vocab_id) > 5
    ),
    created_date = NOW()
    WHERE user_id = owner_id;
END$$


-- Trigger 11: Cập nhật WordSet.is_hidden khi xử lý ReportedSet
CREATE TRIGGER trg_after_update_reportedset_handle_wordset
AFTER UPDATE ON ReportedSet
FOR EACH ROW
BEGIN
    IF NEW.report_status IN (1, 2) THEN
        IF NEW.report_status = 1 THEN
            UPDATE WordSet SET is_hidden = 1 WHERE word_set_id = NEW.word_set_id;
        ELSE
            UPDATE WordSet SET is_hidden = 2 WHERE word_set_id = NEW.word_set_id;
        END IF;
    END IF;
END$$
